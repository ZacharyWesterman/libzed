453 template <typename U = T, typename std::enable_if<is_iterator<U>::value, int>::type = 0>
454 auto flatten() noexcept {
455 struct unchunkData {
456 std::optional<T> data;
457 std::optional<iterator_value<T>> iter;
458 generator<T, S> gen;
459 };
460
461 auto unchunkFn = [](unchunkData &state) -> std::optional<dereference<T>> {
462 // If the generator has not been initialized,
463 // or there is no more data in the chunk,
464 // then try to generate another chunk.
465 if (!state.data || (state.iter.value() == state.data.value().end())) {
466 // If there's nothing left, we're done.
467 auto chunkData = state.gen.next();
468 if (!chunkData) {
469 return {};
470 }
471
472 auto &val = chunkData.value();
473 // Make sure we didn't just get an empty iterator.
474 if (val.begin() == val.end()) {
475 return {};
476 }
477
478 state.data = chunkData;
479 state.iter = state.data.value().begin();
480 }
481
482 // At this point the generator has been initialized, and current chunk has data in it,
483 // so spit out the next value on the chunk.
484
485 return *state.iter.value()++;
486 };
487
488 return generator<dereference<T>, unchunkData>({{}, {}, *this}, unchunkFn);
489 }
