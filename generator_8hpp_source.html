582 template <typename U = T, typename std::enable_if<is_iterator<U>::value, int>::type = 0>
583 auto unchunk() noexcept {
584 struct unchunkData {
585 std::optional<T> data;
586 std::optional<iterator_value<T>> iter;
587 generator<T, S> gen;
588 };
589
590 auto unchunkFn = [](unchunkData &state) -> std::optional<dereference<T>> {
591 // If the generator has not been initialized,
592 // or there is no more data in the chunk,
593 // then try to generate another chunk.
594 if (!state.data || (state.iter.value() == state.data.value().end())) {
595 // If there's nothing left, we're done.
596 auto chunkData = state.gen.next();
597 if (!chunkData) {
598 return {};
599 }
600
601 auto &val = chunkData.value();
602 // Make sure we didn't just get an empty iterator.
603 if (val.begin() == val.end()) {
604 return {};
605 }
606
607 state.data = chunkData;
608 state.iter = state.data.value().begin();
609 }
610
611 // At this point the generator has been initialized, and current chunk has data in it,
612 // so spit out the next value on the chunk.
613
614 return *state.iter.value()++;
615 };
616
617 return generator<dereference<T>, unchunkData>({{}, {}, *this}, unchunkFn);
618 }
