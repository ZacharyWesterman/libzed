97  }

860  //template for string evaluation functions
861  // both ignore spaces, and returns 0.0 if non-numerical characters are present
862  template <typename CHAR>
863  zFloat string<CHAR>::value(int base) const
864  {
865 
866  zFloat value = 0.0;
867 
868  int start = 0;
869  bool isNegative = false;
870 
871  if (string_array[0] == (CHAR)45) //'-' character
872  {
873  start = 1;
874  isNegative = true;
875  }
876 
877  bool pastDecimal = false;
878  bool pastExponent = false;
879  zFloat fracMult = 1.0;
880  int exponent = 0;
881  bool expNegative = false;
882 
883  int length = array_length - 1;
884 
885  for (int i=start; i<length; i++)
886  {
887  //in the rare case that we encounter a null character
888  if (string_array[i] == (CHAR)0)
889  break;
890 
891  //only one decimal point is allowed.
892  //any more, and the string is invalid
893  if (string_array[i] == (CHAR)46) //'.' character
894  {
895  if (pastDecimal)
896  {
897  return 0.0;
898  }
899 
900  pastDecimal = true;
901  }
902 
903  //if a character is not part of a valid number,
904  //the string evaluates to 0.0 and return.
905  else if (isNumeric(string_array[i], base))
906  {
907  if (pastExponent)
908  {
909  exponent *= base;
910  exponent += numeralValue(string_array[i]);
911  }
912  else if (pastDecimal)
913  {
914  fracMult /= (zFloat)base;
915  value += fracMult * (zFloat)numeralValue(string_array[i]); //actual value from character
916  }
917  else
918  {
919  value *= (zFloat)base;
920  value += (zFloat)numeralValue(string_array[i]); //actual value from character
921  }
922  }
923  else if (string_array[i] == (CHAR)69)
924  {
925  pastExponent = true;
926  }
927  else if (pastExponent && (string_array[i] == (CHAR)45))
928  {
929  expNegative = true;
930  }
931  else if (string_array[i] != (CHAR)46)
932  {
933  return 0.0;
934  }
935  }
936 
937 
938  if (exponent)
939  {
940  long valMult = 1;
941 
942  for(int i=0; i<exponent; i++)
943  valMult *= base;
944 
945  if (expNegative)
946  value *= (1 / (zFloat)valMult);
947  else
948  value *= valMult;
949  }
950 
951  //never output "-0"
952  if (value == -0)
953  return 0;
954 
955  if (isNegative)
956  {
957  return -value;
958  }
959  else
960  {
961  return value;
962  }
963  }
