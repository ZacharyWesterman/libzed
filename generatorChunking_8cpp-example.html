// This example shows off generators' ability to un-chunk chunked data.
#include <iostream>
#include <z/all.hpp>
using z::core::join;
using z::core::range;
int main() {
for (auto chunk : range(10).chunk(5)) {
std::cout << '[' << join(chunk, ",") << ']' << std::endl;
}
auto gen = range(10).chunk(5);
for (auto item : gen.unchunk()) {
std::cout << item << std::endl;
}
}
